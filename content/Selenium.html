from selenium import webdriver as wd
from selenium.webdriver.common.keys import Keys as k
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait as wdw
from selenium.webdriver.support import expected_conditions as EC
import time as t
path="C:/chromedriver.exe"
#執行檔
driver=wd.Chrome(path)
driver.get("https://www.dcard.tw/f")
#網站
search=driver.find_element_by_name("query")
#收尋欄的name為query 可以以其他代替
search.clear
#清除收尋欄資訊
search.send_keys("比特幣")
#在收尋欄上輸入 比特幣
search.send_keys(k.RETURN)
#收尋欄 收尋
wdw(driver, 10).until(
        EC.presence_of_element_located((By.CLASS_NAME, "sc-1s9033j-0"))
    )
#等待直到網頁出現sc-1s9033j-0
titles=driver.find_elements_by_class_name("tgn9uw-3")
for title in titles:
    print(title.text)
link=driver.find_element_by_link_text("谨防上当受骗")
link.click()
#點擊此內容
driver.back()
#回到上一頁
driver.back()
driver.forward()
#回到下一頁
t.sleep(10)
driver.quit()

from selenium import webdriver as wd
from selenium.webdriver.common.keys import Keys as k
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait as wdw
from selenium.webdriver.support import expected_conditions as EC
import time as t
import os
import wget
path="C:/chromedriver.exe"
#執行檔
driver=wd.Chrome(path)
driver.get("https://www.instagram.com/")
#網站
username=wdw(driver,10).until(
    EC.presence_of_element_located((By.NAME,"username"))
)
#因為跳轉所以使webdriverwait
password=wdw(driver,10).until(
    EC.presence_of_element_located((By.NAME,"password"))
)
login=driver.find_element_by_xpath('//*[@id="loginForm"]/div/div[3]/button/div')
#使用XPath 登入按鈕
username.clear()
password.clear()
username.send_keys("your-id")
#id
password.send_keys("your-password")
#password
login.click()
#登入
search=wdw(driver,10).until(
    EC.presence_of_element_located((By.XPATH,'//*[@id="react-root"]/section/nav/div[2]/div/div/div[2]/input'))
)
#使用XPath 收尋欄
search.clear()
keyword="#dog"
search.send_keys(keyword)
#收尋欄輸入資料
t.sleep(1)
search.send_keys(k.RETURN)
#收尋欄 收尋
t.sleep(1)
search.send_keys(k.RETURN)
#需要點擊兩次
wdw(driver,10).until(
    EC.presence_of_element_located((By.CLASS_NAME,"FFVAD"))
)
#跳轉使用webdriverwait
for i in range(5):
    driver.execute_script("window.scrollTo(0,document.body.scrollHeight);")
    t.sleep(5)
#使用HTNL檔 把網頁往下展開5次 為了更多資料
imgs=driver.find_elements_by_class_name("FFVAD")
#圖檔(s)
path=os.path.join(keyword)
os.mkdir(path)
#創建檔案 以keyword命名
count=0
for img in imgs:
    save_as=os.path.join(path,keyword+str(count)+".jpg")
    #將檔案命名為keyword 0+count.jpg
    wget.download(img.get_attribute("src"),save_as)
    #自動下載
    count+=1
